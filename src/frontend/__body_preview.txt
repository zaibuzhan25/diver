<!-- Chart.js：国内 CDN + 失败静默处理，不影响页面其余功能 -->
<script>
(function(){
  var cdns = [
    'https://cdn.bootcdn.net/ajax/libs/Chart.js/4.4.1/chart.umd.min.js',
    'https://cdn.jsdelivr.net/npm/chart.js@4.4.1/dist/chart.umd.min.js'
  ];
  var ok = false;
  function load(i){
    if(ok||i>=cdns.length) return;
    var s=document.createElement('script');
    s.src=cdns[i];
    s.onload=function(){ok=true;console.log('[Chart.js] loaded from CDN '+(i+1))};
    s.onerror=function(){console.warn('[Chart.js] CDN '+(i+1)+' failed, trying next...'); load(i+1)};
    document.head.appendChild(s);
  }
  load(0);
})();
</script>
<script>

    if(typeof Chart!=='undefined'){res(Chart);return}
    setTimeout(check,100);
  })();
});
</script>
